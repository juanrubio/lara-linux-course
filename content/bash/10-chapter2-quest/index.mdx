## Part 5: Add Colors (Bonus!)

Make your menu colorful and professional:

<TerminalDemo>
{`$ cat > control_center.sh << 'EOF'
#!/bin/bash

# Color codes
RED='\\033[0;31m'
GREEN='\\033[0;32m'
YELLOW='\\033[1;33m'
BLUE='\\033[0;34m'
CYAN='\\033[0;36m'
NC='\\033[0m' # No Color

while true
do
    clear
    echo -e "${CYAN}========================================${NC}"
    echo -e "${CYAN}       SYSTEM CONTROL CENTER${NC}"
    echo -e "${CYAN}========================================${NC}"
    echo -e "${GREEN}1.${NC} System Information"
    echo -e "${GREEN}2.${NC} Disk Usage"
    echo -e "${GREEN}3.${NC} Memory Usage"
    echo -e "${GREEN}4.${NC} Current Processes"
    echo -e "${GREEN}5.${NC} Network Info"
    echo -e "${GREEN}6.${NC} File Search"
    echo -e "${GREEN}7.${NC} Create Backup"
    echo -e "${RED}8.${NC} Exit"
    echo -e "${CYAN}========================================${NC}"

    read -p "Enter your choice (1-8): " choice
    echo

    case $choice in
        1)
            echo -e "${YELLOW}--- SYSTEM INFORMATION ---${NC}"
            echo "Hostname: $(hostname)"
            echo "Username: $(whoami)"
            echo "Current Date: $(date)"
            echo "Uptime: $(uptime -p)"
            echo "Kernel: $(uname -r)"
            ;;
        2)
            echo -e "${YELLOW}--- DISK USAGE ---${NC}"
            df -h
            echo
            disk_usage=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')
            if [ $disk_usage -gt 80 ]; then
                echo -e "${RED}⚠ Warning: Disk is $disk_usage% full!${NC}"
            else
                echo -e "${GREEN}✓ Disk usage: $disk_usage%${NC}"
            fi
            ;;
        3)
            echo -e "${YELLOW}--- MEMORY USAGE ---${NC}"
            free -h
            ;;
        4)
            echo -e "${YELLOW}--- TOP 10 PROCESSES ---${NC}"
            ps aux --sort=-%mem | head -11
            ;;
        5)
            echo -e "${YELLOW}--- NETWORK INFORMATION ---${NC}"
            echo "IP Address: $(hostname -I)"
            echo "Hostname: $(hostname)"
            ;;
        6)
            read -p "Enter filename to search: " filename
            echo -e "${YELLOW}Searching for: $filename${NC}"
            results=$(find $HOME -name "$filename" -type f 2>/dev/null)
            if [ -z "$results" ]; then
                echo -e "${RED}✗ No files found${NC}"
            else
                echo -e "${GREEN}✓ Found files:${NC}"
                echo "$results"
            fi
            ;;
        7)
            timestamp=$(date +%Y%m%d_%H%M%S)
            backup_file="backup_$timestamp.tar.gz"
            echo -e "${YELLOW}Creating backup: $backup_file${NC}"

            # Create test backup of a small directory
            if [ -d "$HOME/Documents" ]; then
                tar -czf "$HOME/$backup_file" "$HOME/Documents" 2>/dev/null
            else
                mkdir -p "$HOME/test_backup"
                echo "Test file" > "$HOME/test_backup/test.txt"
                tar -czf "$HOME/$backup_file" "$HOME/test_backup" 2>/dev/null
            fi

            if [ -f "$HOME/$backup_file" ]; then
                size=$(du -h "$HOME/$backup_file" | cut -f1)
                echo -e "${GREEN}✓ Backup created successfully!${NC}"
                echo "  File: $backup_file"
                echo "  Size: $size"
                echo "  Location: $HOME/$backup_file"
            else
                echo -e "${RED}✗ Backup failed!${NC}"
            fi
            ;;
        8)
            echo -e "${GREEN}Thank you for using System Control Center!${NC}"
            echo -e "${CYAN}Goodbye!${NC}"
            exit 0
            ;;
        *)
            echo -e "${RED}✗ Invalid choice! Please enter 1-8.${NC}"
            ;;
    esac

    echo
    read -p "Press Enter to continue..."
done
EOF

$ chmod +x control_center.sh
$ ./control_center.sh`}
</TerminalDemo>
