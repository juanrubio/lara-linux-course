## Example Starter Code

Here's a skeleton to get you started. **You must expand this significantly!**

<TerminalDemo>
```python
import json

# Game world data
GAME_WORLD = {
    "village": {
        "name": "Peaceful Village",
        "description": "A cozy village where your adventure begins. You see a path to the forest.",
        "exits": {"north": "forest", "east": "cave"},
        "items": ["bread", "map"]
    },
    "forest": {
        "name": "Dark Forest",
        "description": "Twisted trees loom over you. Something rustles in the shadows.",
        "exits": {"south": "village", "north": "mountain"},
        "items": ["potion"]
    },
    # ADD MORE LOCATIONS!
}

def create_new_player():
    """Create a new player character"""
    print("\n=== CHARACTER CREATION ===")
    name = input("Enter your hero's name: ")

    player = {
        "name": name,
        "health": 100,
        "max_health": 100,
        "inventory": [],
        "location": "village",
        "gold": 50
    }

    return player

def save_game(player):
    """Save player data to file"""
    with open("savegame.json", "w") as file:
        json.dump(player, file, indent=4)
    print("\n*** Game saved! ***")

def load_game():
    """Load player data from file"""
    try:
        with open("savegame.json", "r") as file:
            player = json.load(file)
        print(f"\n*** Welcome back, {player['name']}! ***")
        return player
    except FileNotFoundError:
        return None

def show_location(player):
    """Display current location"""
    loc = GAME_WORLD[player["location"]]

    print("\n" + "="*50)
    print(f"üìç {loc['name']}")
    print("="*50)
    print(loc["description"])

    if loc["items"]:
        print(f"\nüîç You see: {', '.join(loc['items'])}")

    print(f"\nüö™ Exits: {', '.join(loc['exits'].keys())}")

def show_stats(player):
    """Display player stats"""
    print(f"\n‚ù§Ô∏è  Health: {player['health']}/{player['max_health']}")
    print(f"üí∞ Gold: {player['gold']}")
    print(f"üéí Inventory: {', '.join(player['inventory']) if player['inventory'] else 'Empty'}")

def move(player, direction):
    """Move to a new location"""
    loc = GAME_WORLD[player["location"]]

    if direction in loc["exits"]:
        player["location"] = loc["exits"][direction]
        show_location(player)
    else:
        print(f"\n‚ùå You can't go {direction} from here!")

def pickup(player, item):
    """Pick up an item"""
    loc = GAME_WORLD[player["location"]]

    if item in loc["items"]:
        player["inventory"].append(item)
        loc["items"].remove(item)
        print(f"\n‚úì Picked up: {item}")
    else:
        print(f"\n‚ùå There's no {item} here!")

def show_help():
    """Display available commands"""
    print("\n=== COMMANDS ===")
    print("look - Look around")
    print("go <direction> - Move (e.g., 'go north')")
    print("take <item> - Pick up item")
    print("inventory - Show inventory")
    print("stats - Show your stats")
    print("save - Save game")
    print("quit - Exit game")

def main():
    """Main game loop"""
    print("="*50)
    print("     üéÆ ADVENTURE QUEST üéÆ")
    print("="*50)

    # New game or load?
    choice = input("\n1. New Game\n2. Load Game\nChoose: ")

    if choice == "2":
        player = load_game()
        if not player:
            print("No save file found. Starting new game.")
            player = create_new_player()
    else:
        player = create_new_player()

    # Show starting location
    show_location(player)

    # Main game loop
    while True:
        command = input("\n> ").lower().strip()

        if command == "quit":
            save = input("Save before quitting? (y/n): ")
            if save.lower() == "y":
                save_game(player)
            print("\nThanks for playing! üëã")
            break

        elif command == "look":
            show_location(player)

        elif command == "stats":
            show_stats(player)

        elif command == "inventory":
            if player["inventory"]:
                print(f"\nüéí Inventory: {', '.join(player['inventory'])}")
            else:
                print("\nüéí Your inventory is empty!")

        elif command.startswith("go "):
            direction = command.split()[1]
            move(player, direction)

        elif command.startswith("take "):
            item = command.split()[1]
            pickup(player, item)

        elif command == "save":
            save_game(player)

        elif command == "help":
            show_help()

        else:
            print("‚ùì Unknown command. Type 'help' for commands.")

# Start the game!
if __name__ == "__main__":
    main()
```

### 1. Add Combat System

```python
def combat(player, enemy_name):
    """Simple turn-based combat"""
    enemies = {
        "goblin": {"health": 30, "attack": 5, "gold": 10},
        "dragon": {"health": 100, "attack": 20, "gold": 100}
    }

    enemy = enemies[enemy_name].copy()

    print(f"\n‚öîÔ∏è  A {enemy_name} appears!")

    while enemy["health"] > 0 and player["health"] > 0:
        print(f"\nYour HP: {player['health']} | {enemy_name} HP: {enemy['health']}")
        print("1. Attack")
        print("2. Use Potion")
        print("3. Run")

        choice = input("Choose: ")

        if choice == "1":
            damage = 10  # Player attack
            enemy["health"] -= damage
            print(f"You hit for {damage} damage!")

            if enemy["health"] > 0:
                player["health"] -= enemy["attack"]
                print(f"{enemy_name} hits you for {enemy['attack']} damage!")
        # Add more options...

    if player["health"] <= 0:
        print("\nüíÄ You died! Game Over.")
        return False
    else:
        print(f"\nüéâ You defeated the {enemy_name}!")
        player["gold"] += enemy["gold"]
        return True
```

## Testing Your Game

Before you call it complete, test these scenarios:

1. **Fresh Start Test**
   - Start new game
   - Play through to the end
   - Make sure it's winnable!

2. **Save/Load Test**
   - Play partway
   - Save and exit
   - Load game
   - Continue where you left off

3. **Edge Cases**
   - Try invalid commands
   - Try to go invalid directions
   - Try to pick up items that don't exist
   - Test with empty inventory

4. **Death Test**
   - Make sure game over works
   - What happens if player dies?

5. **Win Test**
   - Complete the game
   - Is the ending satisfying?
