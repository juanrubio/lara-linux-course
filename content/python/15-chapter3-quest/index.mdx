elif choice == "3":
    print("\\n--- COMPLETE A TASK ---")
    if not tasks:
        print("No tasks to complete!")
    else:
        # Show all tasks first
        for index, task in enumerate(tasks, start=1):
            print(f"{index}. {task}")

        try:
            task_num = int(input("\\nWhich task number? "))
            if 1 <= task_num <= len(tasks):
                completed = tasks.pop(task_num - 1)  # -1 because list is 0-indexed
                print(f"✓ Completed: {completed}")
            else:
                print("Invalid task number!")
        except ValueError:
            print("Please enter a number!")`}
</TerminalDemo>

<StoryBox character="python_pete">
  Notice the try/except block? This is called error handling!

  If the user types "abc" instead of a number, int() would crash. But try/except catches the error and prints a friendly message instead!

  This makes your program more professional and user-friendly!
</StoryBox>

## The Complete Code

Here's what your full program should look like:

<TerminalDemo>
{`# Todo List Manager - Complete Version
tasks = []
running = True

print("=== WELCOME TO YOUR TODO LIST ===")
print()

while running:
    print("\\n--- MAIN MENU ---")
    print("1. Add a task")
    print("2. View all tasks")
    print("3. Complete a task")
    print("4. View stats")
    print("5. Quit")
    print()

    choice = input("Choose an option (1-5): ")

    if choice == "1":
        # ADD TASK
        print("\\n--- ADD A TASK ---")
        task = input("Enter a task: ")
        if task:
            tasks.append(task)
            print(f"✓ Added: {task}")
        else:
            print("Task cannot be empty!")

    elif choice == "2":
        # VIEW TASKS
        print("\\n--- YOUR TASKS ---")
        if not tasks:
            print("No tasks yet! Add some to get started.")
        else:
            for index, task in enumerate(tasks, start=1):
                print(f"{index}. {task}")
            print(f"\\nTotal tasks: {len(tasks)}")

    elif choice == "3":
        # COMPLETE TASK
        print("\\n--- COMPLETE A TASK ---")
        if not tasks:
            print("No tasks to complete!")
        else:
            for index, task in enumerate(tasks, start=1):
                print(f"{index}. {task}")

            try:
                task_num = int(input("\\nWhich task number? "))
                if 1 <= task_num <= len(tasks):
                    completed = tasks.pop(task_num - 1)
                    print(f"✓ Completed: {completed}")
                else:
                    print("Invalid task number!")
            except ValueError:
                print("Please enter a number!")

    elif choice == "4":
        # STATS
        print("\\n--- TASK STATISTICS ---")
        total = len(tasks)
        print(f"Total tasks: {total}")

        if total == 0:
            print("Great job! You've completed everything!")
        elif total <= 3:
            print("You're almost done! Keep going!")
        else:
            print(f"You have {total} tasks to tackle. You got this!")

    elif choice == "5":
        # QUIT
        print("\\nThanks for using Todo List Manager!")
        if not tasks:
            print("You finished all your tasks!")
        else:
            print(f"You still have {len(tasks)} tasks pending.")
        running = False

    else:
        print("Invalid choice! Please choose 1-5.")

print("\\nGoodbye! Come back soon!")`}
</TerminalDemo>
